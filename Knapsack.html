التحليل الهيكلي لخوارزميات البرمجة التنافسية

مسألة: حقيبة الظهر (Knapsack 0/1)
هذا المستند يحلل مسألة حقيبة الظهر الكلاسيكية (0/1 Knapsack DP). عند الضغط على أمثلة مشابهة، سيتغير الكود البرمجي أمامك ليوضح كيف تختلف كتابة هذا الجزء لجعله أكثر ديناميكية أو ليناسب قيوداً مختلفة.
1

المتغيرات وحالة المسألة (State)

#include <iostream>
using namespace std;

long long dp[200][200000];
int v[200], w[200];
كود معدل: مسألة الحقيبة غير المحدودة (Unbounded Knapsack)
#include <iostream>
using namespace std;

// يمكننا أخذ العنصر مرات لا نهائية، لا نحتاج لتتبع الـ Index
// المصفوفة تصبح أحادية البعد (1D Array) للحجم فقط!
long long dp[200000]; 
int v[200], w[200];
نستخدم long long لمصفوفة dp لأن مجموع قيم العناصر المأخوذة (Values) قد يتجاوز الحد الأقصى للمتغير int. المصفوفة ثنائية الأبعاد dp[i][capacity] تمثل: "أقصى قيمة يمكن تحقيقها باستخدام العناصر من 0 إلى i بوزن متبقي capacity".
تغير المعطيات
في مسألة الحقيبة غير المحدودة (Unbounded Knapsack) حيث يمكنك أخذ نفس العنصر أكثر من مرة، لا نحتاج لتذكر "عند أي عنصر نحن الآن"، فالحالة تعتمد فقط على "كم تبقى من الوزن؟". لذلك يتم الاستغناء عن البعد الأول وتتحول المصفوفة لـ 1D.

الحالات الأساسية (Base Cases)

long long f (int i, int capacity) {

    if (i == 0) {
        if (capacity < w[i]) {
            return 0;
        }
        else {
            return v[i];
        }
    }
كود معدل: تبسيط الحالات الأساسية
long long f (int i, int capacity) {

    // الاستغناء عن التحقق من i == 0 بأسلوب أنظف وأشمل
    if (i < 0) {
        return 0; // لم يتبق عناصر لنأخذها
    }
    
    // (ملاحظة: التحقق من capacity >= w[i] سيتم في الأسفل)
عندما نصل للعنصر الأول (الفهرس 0)، فإننا مجبرون على اتخاذ قرار نهائي بشأنه لأنه لم يعد هناك عناصر أخرى لنستدعيها.
إذا كانت الحقيبة تتسع لوزنه، نأخذه ونرجع قيمته. إذا كانت لا تتسع، نتركه ونرجع 0.
تغير المعطيات
في الأكواد الاحترافية (Clean Code)، يتم غالباً تبسيط شرط التوقف. بدلاً من التعامل الخاص مع العنصر 0، نسمح للدالة باستدعاء f(-1, cap)، ويكون الـ Base Case هو: "إذا أصبحت i < 0 فهذا يعني انتهاء العناصر، أرجع صفر". هذا يقلل من تكرار الشروط البرمجية.

تقنية الحفظ (Memoization)

    if (dp[i][capacity] != -1) {
        return dp[i][capacity];
    }
كود معدل: مسائل البحث عن الحد الأدنى (Minimization)
    // في مسائل الـ (Minimum) مثل Coin Change
    // نهيئ المصفوفة برقم ضخم كـ 1e18، ونتحقق بناءً عليه
    if (dp[i][capacity] != 1e18) {
        return dp[i][capacity];
    }
الخطوة الأهم لمنع Time Limit Exceeded (TLE). إذا كانت الحالة الحالية (العنصر i بوزن متبقي capacity) قد حُسبت مسبقاً وتم تغيير قيمتها من -1، نعيد القيمة فوراً دون استدعاءات إضافية.
تغير المعطيات
في مسألتنا نبحث عن الحد الأقصى (Maximize Value) لذلك استخدمنا -1 كقيمة أولية.
أما في مسائل مثل أقل عدد عملات لتكوين مبلغ (Coin Change - Min)، سنقوم بتهيئة المصفوفة برقم كبير جداً (Infinity) لكي ينجح تطبيق دالة الـ min، وسيصبح شرط التحقق مطابقاً للكود المجاور.

معادلة الانتقال (Transition) - خياران فقط

    // i في حالة إستبعاد العنصر 
    dp[i][capacity] = f(i - 1, capacity);

    // i في حالة أخذ العنصر (يجب التحقق من السعة أولاً)
    if (capacity >= w[i]) {
        dp[i][capacity] = max(
            dp[i][capacity], 
            f(i - 1, capacity - w[i]) + v[i]
        );
    }

    return dp[i][capacity];
}
كود معدل: حساب "عدد الطرق" الممكنة بدلاً من الحد الأقصى
    // استبعاد العنصر
    long long ways = f(i - 1, capacity);

    // في حالة حساب عدد الطرق (Ways)، نستبدل max بعملية الجمع +
    if (capacity >= w[i]) {
        ways = (ways + f(i - 1, capacity - w[i])) % MOD;
    }

    dp[i][capacity] = ways;
    return dp[i][capacity];
}
هذا هو الجوهر الكامل لمسألة الحقيبة (Take or Leave). لكل عنصر خياران:
1. استبعاده: ننتقل للعنصر i-1 ونحتفظ بنفس السعة المتبقية.
2. أخذه: إذا كانت الحقيبة تتسع لوزنه، ننتقل للعنصر i-1، ونخصم وزنه من السعة، ونجمع قيمته v[i] للناتج.
نختار المسار الذي يُعطي القيمة الأكبر (max).
تغير المعطيات
إذا تغير صيغة السؤال إلى: "بكم طريقة مختلفة يمكننا ملء الحقيبة بوزن محدد بالضبط؟"، فلن نبحث عن الحد الأقصى max. بل سنستخدم عملية الجمع + لنجمع عدد الطرق الناتجة عن "استبعاد العنصر" مع عدد الطرق الناتجة عن "أخذ العنصر" (مع تطبيق Modulo إذا كانت الأرقام ضخمة).

التهيئة والدالة الرئيسية (Main)

int main() {
  
    // تهيئة المصفوفة ثنائية الأبعاد بـ 1-
    for (int i = 0; i < 200; i++) {
    for (int j = 0; j < 200000; j++) {
          dp[i][j] = -1;
    }
    }

    int n, weight;
    cin >> n >> weight;
  
    for (int i = 0; i < n; i++) {
        cin >> w[i] >> v[i];
    }

    // استدعاء الدالة لآخر عنصر بكامل سعة الحقيبة
    cout << f(n - 1, weight) << endl;
}
كود معدل: التهيئة السريعة واستخدام Bottom-Up 1D
#include <cstring> // من أجل memset

int main() {
    /* ... (قراءة المعطيات) ... */

    // تحسين المساحة (Space Optimization) لـ O(W)
    long long dp[200005] = {0};
    
    for (int i = 0; i < n; i++) {
        // يجب المرور بشكل عكسي (من النهاية للبداية) في حالة 0/1
        for (int j = weight; j >= w[i]; j--) {
            dp[j] = max(dp[j], dp[j - w[i]] + v[i]);
        }
    }

    cout << dp[weight] << endl;
}
نهيئ مصفوفة الحفظ بالقيمة -1 عبر حلقتين متداخلتين. ثم نقوم بقراءة عدد العناصر n والوزن الأقصى weight، تليها أوزان وقيم كل عنصر. في النهاية نطبع ناتج استدعاء الدالة الرئيسية التي تبدأ من العنصر الأخير وتتراجع.
تغير المعطيات
في المسائل التي تعاني من ضيق شديد في الذاكرة (Memory Limit)، يتم الاستغناء تماماً عن المصفوفة ثنائية الأبعاد والدالة التراجعية! بدلاً من ذلك، نستخدم أسلوب Bottom-Up بمصفوفة أحادية البعد 1D يتم تحديثها بشكل عكسي مع كل عنصر جديد لتوفير الذاكرة لتصبح O(W) بدلاً من O(N*W).
