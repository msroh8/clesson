التحليل الهيكلي لخوارزميات البرمجة التنافسية

مسألة: مسارات الشبكة (Grid Paths 2D DP)
هذا المستند يحلل مسألة تعتمد على البرمجة الديناميكية في الشبكات (2D Grid DP) لحساب عدد الطرق الممكنة. عند الضغط على أمثلة مشابهة، سيتغير الكود ليوضح كيف يمكن تكييف هذا الأسلوب لمسائل أخرى مثل البحث عن الحد الأقصى أو تحسين استهلاك الذاكرة.

الإعدادات وقراءة الشبكة (Input Handling)

#include <iostream>
#include <vector>
using namespace std;

const long long MOD = 1000000007LL;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;

    vector<string> grid(n);
    for (int i = 0; i < n; ++i) {
        cin >> grid[i];
    }
كود معدل: شبكة أرقام (Weighted Grid)
// إذا كانت الشبكة تحتوي على أرقام (مثل تكلفة أو نقاط) وليس رموز '*' و '.'
// نستخدم vector ثنائي الأبعاد للأرقام بدلاً من vector of strings

int n, m; // الشبكة قد تكون N x M
cin >> n >> m;

vector<vector<int>> grid(n, vector<int>(m));
for (int i = 0; i < n; ++i) {
    for (int j = 0; j < m; ++j) {
        cin >> grid[i][j];
    }
}
نعرّف المتغير MOD لأن أعداد الطرق تنمو أُسياً وقد تتجاوز سعة البيانات بسهولة. بما أن الشبكة معطاة كحروف متصلة (مثال: ..*..)، نستخدم vector<string> لقراءتها. هذا أسرع بكثير وأكثر كفاءة من قراءة كل حرف بمفرده في مصفوفة ثنائية الأبعاد للحروف.
تغير المعطيات
إذا كانت المسألة تطلب إيجاد "أقل مسار تكلفة" (Min Path Sum)، فالشبكة ستتكون من أرقام منفصلة بالمسافات بدلاً من الرموز. هنا يجب التغيير من vector<string> إلى مصفوفة ثنائية الأبعاد من نوع int أو long long وقراءتها بحلقتين متداخلتين كما يظهر في الكود.

تهيئة الـ DP (State Initialization)

    // dp[i][j] = number of ways to reach (i, j)
    vector<vector<long long>> dp(n, vector<long long>(n, 0));
كود معدل: تحسين استهلاك الذاكرة (Space Optimization)
    // بدلاً من مصفوفة ثنائية تستهلك O(N^2)
    // نستخدم صفين فقط O(N) لأننا نحتاج الصف الحالي والسابق فقط!
    
    vector<long long> prev_dp(n, 0);
    vector<long long> curr_dp(n, 0);
هنا نعرف مصفوفة ثنائية الأبعاد بأحجام N x N ونهيئ جميع خلاياها بالرقم 0. الخلية dp[i][j] ستحتفظ بعدد الطرق الممكنة للوصول إلى الصف i والعمود j من نقطة البداية (0,0).
تغير المعطيات
إذا كانت شبكة Grid عملاقة (مثلاً 10^5 x 10^5)، فإن مصفوفة ثنائية الأبعاد ستسبب خطأ امتلاء الذاكرة Memory Limit Exceeded. بما أن حساب الصف الحالي يعتمد فقط على الصف الذي فوقه مباشرة، نقوم بتحسين المساحة (Space Optimization) باستخدام مصفوفتين أحاديتين فقط (واحدة للصف السابق وواحدة للصف الحالي) ويتم التبديل بينهما.

كود معدل: تجنب شروط الحدود (Padding Technique)
// لتبسيط الشروط المعقدة بالداخل، نبدأ من 1 بدلاً من 0
// ونكون قد قمنا بتهيئة حجم المصفوفة كـ (n+1) x (n+1)
for (int i = 1; i <= n; ++i) {
    for (int j = 1; j <= n; ++j) {
        // grid is 0-indexed, so we check i-1, j-1
        if (grid[i-1][j-1] == '*') {
            dp[i][j] = 0;
            continue;
        }
نمر على كل خلية في الشبكة. إذا كانت الخلية الحالية تحتوي على النجمة '*'، فهذا يعني أنها "فخ" أو عائق. لا يمكن المرور من خلالها، لذا نُثبت عدد الطرق للوصول إليها بـ 0 ونستخدم continue لتخطي باقي الحسابات لهذه الخلية فوراً.
تغير المعطيات
أحد أساليب البرمجة التنافسية الذكية هو الحشو (1-Based Indexing Padding). لكي نتجنب كتابة الشروط المزعجة مثل i > 0 لاحقاً (خوفاً من الخروج عن حدود المصفوفة السالبة)، نقوم ببدء الحلقات من 1 ونجعل حجم مصفوفة الـ dp يزيد بمقدار 1. الصف 0 والعمود 0 سيكونا أصفاراً تلقائياً ويمثلان جداراً وهمياً آمناً!

معادلة الانتقال (Transition Equation)

            if (i == 0 && j == 0) {
                dp[i][j] = 1; // starting cell (if not trap)
            } else {
                long long from_up   = (i > 0 ? dp[i - 1][j] : 0);
                long long from_left = (j > 0 ? dp[i][j - 1] : 0);
                dp[i][j] = (from_up + from_left) % MOD;
            }
        }
    }
كود معدل: البحث عن الحد الأقصى للنقاط (Max Path Sum)
            if (i == 0 && j == 0) {
                dp[i][j] = grid[0][0]; // قيمة الخلية الأولى
            } else {
                // نستبدل الجمع باختيار المسار الأكبر، ثم نجمع قيمة الخلية الحالية
                long long from_up   = (i > 0 ? dp[i - 1][j] : -1e18);
                long long from_left = (j > 0 ? dp[i][j - 1] : -1e18);
                
                dp[i][j] = max(from_up, from_left) + grid[i][j];
            }
الشرط الأساسي: إذا كنا في نقطة البداية (ولم تكن فخاً)، فهناك "طريقة واحدة" للوقوف فيها.
لأي خلية أخرى: يمكننا الوصول إليها إما من الخلية التي فوقها أو الخلية التي على يسارها. عدد الطرق للوصول إليها يساوي (مجموع) طرق الوصول للخلية العلوية والخلية اليسرى. ونأخذ الـ Modulo مباشرة لمنع الـ Overflow.
تغير المعطيات
ماذا لو كانت المسألة تطلب جمع أكبر عدد من النقاط (العملات) أثناء السير، بدلاً من حساب عدد الطرق؟
هنا نستبدل عملية الجمع + بدالة max لاختيار المسار الأفضل بين الأعلى واليسار، ثم نضيف عليه قيمة الخلية الحالية. ونراعي استخدام -1e18 كقيمة للخلايا خارج الحدود لتجنب اختيارها.

كود معدل: تتبع المسار (Path Backtracking)
    // إذا طلب منك طباعة الإحداثيات للمسار الفعلي
    cout << dp[n-1][n-1] << "\n";
    
    int i = n-1, j = n-1;
    while(i > 0 || j > 0) {
        cout << "(" << i << "," << j << ") <- ";
        // نتراجع بناءً على مصفوفة parent تم بناؤها بالخطوة السابقة
        pair<int,int> p = parent[i][j];
        i = p.first; j = p.second;
    }
    cout << "(0,0)\n";
بما أننا نبني الحل من أعلى اليسار (0,0) متجيهن لأسفل ويمين، فإن الهدف النهائي للوصول إليه يتواجد في الزاوية السفلية اليمنى من الشبكة، والتي إحداثياتها هي (n-1, n-1) في حالة القراءة بصفر كبداية الفهارس (0-indexed).
تغير المعطيات
في مسابقات مثل CSES، قد يطلب منك ليس فقط إيجاد العدد، بل طباعة "حركات المسار" (D D R R). لذلك، بجانب مصفوفة الـ dp، ستحتاج لبناء مصفوفة مساعدة مثلاً تُسمى parent لتخزين من أين أتيت لكل خلية. وفي النهاية تقوم بحلقة while تبدأ من نقطة النهاية وتتراجع خطوة بخطوة حتى تصل لنقطة البداية.
